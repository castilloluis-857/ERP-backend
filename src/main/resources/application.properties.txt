# =============================================================================
# ERP - Configuración del Backend Spring Boot

# 1. Base de Datos (MySQL)
# -----------------------------------------------------------------------------
spring.datasource.url=jdbc:mysql://localhost:3306/erp_db?createDatabaseIfNotExist=true&serverTimezone=UTC&useSSL=false&allowPublicKeyRetrieval=true

server.error.include-message=always
